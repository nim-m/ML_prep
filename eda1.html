csv_files <- list.files(pattern = "\\.csv$")
df_vector <- c()

for (csv_file in csv_files) {
  df_name <- sub("\\.csv$", "", csv_file)
  new_df_name <- gsub("-", "_", df_name)
  df <- read.csv(csv_file)
  assign(new_df_name, df, envir = .GlobalEnv)
  df_vector <- c(df_vector, new_df_name)
}
